query getOrderDetails($id: ID!) {
  order(id: $id) {
    id
    name
    createdAt
    # Dados do Cliente
    customer {
      email
      firstName
      lastName
    }
    displayFinancialStatus
    displayFulfillmentStatus
    cancelledAt
    cancelReason
    totalPriceSet {
      shopMoney {
        amount
        currencyCode
      }
    }
    fulfillments {
      id
      status
      trackingInfo {
        number
        url
        company
      }
    }
    lineItems(first: 5) {
      edges {
        node {
          title
          quantity
          sku
        }
      }
    }
  }
}

  customer(id: $id) {
    id
    firstName
    lastName
    email
    # Opcional: Podes também ver as últimas encomendas deste cliente
    orders(first: 5) {
      edges {
        node {
          id
          name
          displayFulfillmentStatus
        }
      }
